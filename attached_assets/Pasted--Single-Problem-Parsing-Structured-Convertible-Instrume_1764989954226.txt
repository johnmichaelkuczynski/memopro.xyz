```python
def process_convertibles(convertibles_list, tentative_ipo_price):
    """
    Process list of convertible instruments.
    Returns: total expected conversion shares (in millions)
    """
    total_conversion_shares = 0
    
    for conv in convertibles_list:
        if conv["type"] == "SAFE":
            # Calculate conversion price: lower of $14.50 or 85% of IPO
            price_option1 = conv["price1"]
            price_option2 = 0.85 * tentative_ipo_price
            conversion_price = min(price_option1, price_option2)
            
            shares = conv["amount"] / conversion_price  # in millions
            total_conversion_shares += shares
            print(f"  SAFE: ${conv['amount']}M → {shares:.3f}M shares @ ${conversion_price:.2f}")
            
        elif conv["type"] == "Debt":
            # Check if condition met: IPO > $15
            if tentative_ipo_price > conv.get("condition", 0):
                shares = conv["amount"] / conv["price"]
                total_conversion_shares += shares
                print(f"  Debt: ${conv['amount']}M → {shares:.3f}M shares @ ${conv['price']:.2f}")
            else:
                print(f"  Debt: ${conv['amount']}M → NOT CONVERTED (IPO ${tentative_ipo_price:.2f} ≤ ${conv.get('condition', 0):.2f})")
                
        elif conv["type"] == "Grant":
            # Probability-weighted shares
            expected_shares = conv["shares"] * conv.get("probability", 1.0)
            total_conversion_shares += expected_shares
            print(f"  Grant: ${conv['amount']}M → {expected_shares:.3f}M shares (expected, {conv.get('probability', 1.0):.0%} probability)")
    
    return total_conversion_shares
```
