```python
def process_contingencies(contingencies_list, tentative_ipo_price, revenue_estimate=None):
    """
    Process list of contingent liabilities.
    Returns: (expected_dilution_shares, expected_cash_cost)
    """
    total_shares = 0
    total_cash_cost = 0  # in millions
    
    for cont in contingencies_list:
        if cont["type"] == "earnout":
            # Check if revenue condition would be met
            # For IPO pricing, we use probability as estimate
            expected_shares = cont["shares"] * cont.get("probability", 0)
            total_shares += expected_shares
            
            print(f"  Earnout: {cont['shares']:.1f}M shares if {cont.get('condition', '')}")
            print(f"    → {expected_shares:.3f}M expected shares ({cont.get('probability', 0):.0%} probability)")
            
        elif cont["type"] == "warrant":
            # Price-based warrant: (IPO - strike) × shares × probability
            expected_shares = cont["shares"] * cont.get("probability", 0)
            total_shares += expected_shares
            
            if tentative_ipo_price > cont.get("strike", 0):
                # Calculate expected cash cost (value given away)
                value_per_share = tentative_ipo_price - cont["strike"]
                cash_cost = value_per_share * expected_shares
                total_cash_cost += cash_cost
                
                print(f"  Warrant: {cont['shares']:.1f}M @ ${cont['strike']:.2f}")
                print(f"    → {expected_shares:.3f}M expected shares")
                print(f"    → ${cash_cost:.2f}M expected cost (in-the-money)")
            else:
                print(f"  Warrant: {cont['shares']:.1f}M @ ${cont['strike']:.2f} (out-of-money)")
                
        elif cont["type"] == "litigation":
            # Simple probability-weighted cash payment
            expected_cost = cont["payment"] * cont.get("probability", 0)
            total_cash_cost += expected_cost
            
            print(f"  Litigation: ${cont['payment']:.1f}M potential payment")
            print(f"    → ${expected_cost:.2f}M expected cost ({cont.get('probability', 0):.0%} probability)")
    
    return total_shares, total_cash_cost
```
